\begin{definitionbox}{Definition (Bad Object)}
\begin{definition}[Bad Object]
\label{thm:bad-object}
A bad object is a Cauchy sequence with no decoration.
\end{definition}
\end{definitionbox}

\begin{theorembox}{Bad Result}
\begin{theorem}[Bad Result]
\label{thm:bad-result}
For all bad objects $x$, there exists a bad object $y$.
\end{theorem}
\end{theorembox}
